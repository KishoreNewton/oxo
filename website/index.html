T

Transform

Mutate log entries before they hit the viewer. Redact PII, normalize fields, enrich with metadata.

pub fn transform(entry: LogEntry) -> LogEntry {
    let mut e = entry;
    // Redact credit card numbers
    e.line = redact_pii(&e.line);
    // Add team ownership label
    e.labels.insert("team", lookup_team(&e));
    e
}
F

Filter

Drop noisy entries before they consume buffer space. Business logic filtering that LogQL can't express.

pub fn filter(entry: &LogEntry) -> bool {
    // Drop health check noise
    if entry.line.contains("/healthz") {
        return false;
    }
    // Keep only prod errors after hours
    is_prod(&entry) && is_after_hours()
}
M

Metrics

Extract custom metrics from log lines. Feed them to the live dashboard without external tooling.

pub fn extract(entry: &LogEntry) -> Vec<Metric> {
    let duration = parse_duration(&entry.line);
    let status = parse_status(&entry.line);
    vec![
        Metric::histogram("latency", duration),
        Metric::counter("requests", status),
    ]
}

Plugins run in a Wasmtime sandbox. They cannot access the filesystem, network, or host memory. A crashing plugin cannot take down the host.

Enable WASM plugins

# Build with WASM support (uses wasmtime runtime)
cargo install oxo --features wasm

# Or in Cargo.toml
[features]
default = ["wasm"]

Get started in 30 seconds

1

Install

cargo install oxo
2

Run

# Demo mode (no infra needed)
oxo

# Connect to Loki
oxo --url http://localhost:3100

# Tail Docker container
oxo --backend docker

# Pipe any command
kubectl logs -f deploy/api | oxo
3

Configure

# ~/.config/oxo/config.toml

[[sources]]
name = "Production"
type = "loki"
url  = "https://loki.prod:3100"
token = "bearer-token"

[[sources]]
name = "Local Logs"
type = "file"
path = "/var/log/app/*.log"

[alerts]
enabled = true

[[alerts.rules]]
name = "Error Spike"
pattern = "(?i)error|panic"
cooldown_secs = 30
